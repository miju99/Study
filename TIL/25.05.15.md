        {
            if (soundPlayerDic.ContainsKey(type))
            {
                foreach (SoundPlayer sp in soundPlayerDic[type])
                {
                    sp.AudioSourceComp.Stop();
                    Destroy(sp.gameObject);
                }
                soundPlayerDic[type].Clear();
            }
        }
