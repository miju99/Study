  - 문자열 검색 (IndexOf)
 
        string str = "Hello, World!";
        int index = str.indexOf("World");
        //H부터 0으로 시작해서 H/e/l/l/o/,/ /W이 7번째 시작하므로 7이 나온다.

  - 문자열 대체 (Replace)
 
        string str = "Hello, World!";
        string newStr = str.Replace("World", "Universe");
        //newStr은 Hello, Universe!, str은 Hello, World!로 나온다.
